def recup_infos_base base_name
  doc = Nokogiri::Slop IO.read("infosconnexion.xml")
  tab = []
  doc.infos.sgbd.each do |sgbd|
    if sgbd["name"] == base_name
      sgbd.param.each do |p|
        tab << [p["name"], p.content]
      end
    end
  end
  return tab
end
